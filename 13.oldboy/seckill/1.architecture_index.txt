功能2:秒级请求限制:
service/antispam.go

type SecLimit struct{
    count int //计数
    curTime int64 //时间
}

//计数:
func (p *SecLimit)Count(nowTime int64) (curCount int){
    if p.curTime != nowTime{//如果传入的时间不是限速记录的当前时间
        p.count=1
        p.curTime= nowTime
    }else{
        p.count++
    }
    return p.count
}

//检查计数:
func (p *SecLimit)Check(nowTime int64) {
    if p.curTime != nowTime{//如果传入的时间不是限速记录的当前时间
        return 0
    }
    return p.count
}
